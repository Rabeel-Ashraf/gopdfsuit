POST

/api/v1/generate/template-pdf

Generate template-based PDF from JSON configuration

Content-Type: application/json

Template Configuration

  • config.page: A4, LETTER, LEGAL, A3, A5
  • config.pageAlignment: 1 (Portrait), 2 (Landscape)
  • config.pageBorder: "left:right:top:bottom" border widths
  • config.watermark: Optional diagonal watermark text
  • props format: "font:size:style:align:left:right:top:bottom"
  • style codes: 3-digit (bold:italic:underline) - 000 to 111
cURL Example 
curl -X POST "http://localhost:8080/api/v1/generate/template-pdf" \
  -H "Content-Type: application/json" \
  -d '{
    "config": {
      "pageBorder": "2:2:2:2",
      "page": "A4",
      "pageAlignment": 1,
      "watermark": "CONFIDENTIAL"
    },
    "title": {
      "props": "font1:20:110:center:0:0:2:0",
      "text": "Multi-Page Document"
    },
    "table": [{
      "maxcolumns": 2,
      "rows": [{
        "row": [{
          "props": "font1:12:100:left:1:1:1:1",
          "text": "Field Name:"
        }, {
          "props": "font1:12:000:left:1:1:1:1",
          "text": "Field Value"
        }]
      }]
    }],
    "footer": {
      "font": "font1:10:001:center",
      "text": "Page footer with auto-pagination"
    }
  }' \
  --output document.pdf

Combine multiple PDF files into a single document

Content-Type: multipart/form-data

Features

Landscape Layout with Python 
import requests
import json

url = "http://localhost:8080/api/v1/generate/template-pdf"
template = {
    "config": {
        "pageBorder": "1:1:1:1",
        "page": "A4",
        "pageAlignment": 2  # Landscape orientation
    },
    "title": {
        "props": "font1:22:111:center:0:0:2:0",
        "text": "Landscape Survey Form"
    },
    "table": [{
        "maxcolumns": 6,  # More columns fit in landscape
        "rows": [{
            "row": [{
                "props": "font1:14:100:left:1:1:1:1",
                "text": "Question 1:"
            }, {
                "props": "font1:12:000:center:1:1:1:1",
                "chequebox": True

                "text": "Excellent"

JavaScript/Node.js Integration

Node.js with Fetch API 
const fs = require('fs');

async function generatePDF() {
    const template = {
        "config": {
            "pageBorder": "1:1:1:1",
            "page": "A4",
            "pageAlignment": 1
        },
        "title": {
            "props": "font1:24:100:center:0:0:1:0",
            "text": "JavaScript Generated Document"
        },
        "table": [{
            "maxcolumns": 2,
            "rows": [{
                "row": [{
                    "props": "font1:12:100:left:1:1:1:1",
                    "text": "Generated with:"
                }, {
                    "props": "font1:12:010:left:1:1:1:1",
                    "text": "Node.js & GoPdfSuit"
                }]
            }]
        }]
    };

    try {
        const response = await fetch('http://localhost:8080/api/v1/generate/template-pdf', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(template)
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const buffer = await response.arrayBuffer();
        fs.writeFileSync('generated-document.pdf', Buffer.from(buffer));
        console.log('PDF generated successfully!');
    } catch (error) {
        console.error('Error generating PDF:', error);
    }
}

generatePDF();
